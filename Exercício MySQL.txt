CREATE  DATABASE  db_generation_game_online ;
USE db_generation_game_online;

CRIAR  TABELA  tb_classe (
    id INT AUTO_INCREMENT,
    CHAVE PRIMÁRIA (id),
    nome VARCHAR ( 20 ) NÃO NULO ,
    evolucoes INT  NOT NULL ,
    tipo VARCHAR ( 15 ) NÃO NULO
) MOTOR = INNODB;

CRIAR  TABELA  tb_personagem (
    id INT AUTO_INCREMENT,
    CHAVE PRIMÁRIA (id),
    nome VARCHAR ( 50 ) NÃO NULO ,
    ataque INT  NOT NULL ,
    defesa INT  NOT NULL ,
    Sabedoria INT  NOT NULL ,
    velocidade INT  NÃO NULO ,
    nivel INT  NOT NULL ,
    id_classe INT  NOT NULL ,
    CHAVE ESTRANGEIRA (id_classe)
        REFERÊNCIAS tb_classe (id)
) MOTOR = INNODB;
